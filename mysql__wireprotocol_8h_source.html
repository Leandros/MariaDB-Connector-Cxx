39 /* Packet handling */
40 #define PACKET_INIT(packet, enum_type, c_type) \
41  { \
42  packet = (c_type) my_mcalloc( packet_methods[enum_type].struct_size, MYF(MY_WME | MY_ZEROFILL)); \
43  ((c_type) (packet))->header.m = &packet_methods[enum_type]; \
44  }
45 #define PACKET_WRITE(packet, conn) ((packet)->header.m->write_to_net((packet), (conn)))
46 #define PACKET_READ(packet, conn) ((packet)->header.m->read_from_net((packet), (conn)))
47 #define PACKET_FREE(packet) ((packet)->header.m->free_mem((packet), FALSE))
48 
49 #define PACKET_INIT_ALLOCA(packet, enum_type) \
50  { \
51  memset(&(packet), 0, packet_methods[enum_type].struct_size); \
52  (packet).header.m = &packet_methods[enum_type]; \
53  }
54 #define PACKET_WRITE_ALLOCA(packet, conn) PACKET_WRITE(&(packet), (conn))
55 #define PACKET_READ_ALLOCA(packet, conn) PACKET_READ(&(packet), (conn))
56 #define PACKET_FREE_ALLOCA(packet) (packet.header.m->free_mem(&(packet), TRUE))
